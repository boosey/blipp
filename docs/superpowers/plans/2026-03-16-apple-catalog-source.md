# Apple Podcasts Catalog Source Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Switch catalog discovery from Podcast Index to Apple Podcasts Charts API with proper category taxonomy, queue-based refresh, and pending deletion workflow.

**Architecture:** New Apple Podcasts client fetches charts across 19 genres + batch iTunes Lookup for metadata/feed URLs. A new queue consumer handles the heavy refresh work (avoiding Worker timeout). Category model with join table enables browse-by-category. Existing Podcast Index source retained as fallback.

**Tech Stack:** Cloudflare Workers, Hono, Prisma 7, React 19, Apple Charts API, iTunes Lookup/Search API

**Spec:** `docs/superpowers/specs/2026-03-16-apple-catalog-source-design.md`

```typescript
/**
 * Client for Apple Podcasts Charts API and iTunes Lookup/Search APIs.
 * No authentication required. Used for catalog discovery.
 */

const CHARTS_BASE = "https://rss.marketingtools.apple.com/api/v2";
const ITUNES_BASE = "https://itunes.apple.com";
const LOOKUP_BATCH_SIZE = 150;
const RETRY_MAX = 3;
const RETRY_BASE_MS = 1000;
const INTER_REQUEST_DELAY_MS = 500;

/** Apple's top-level podcast genre IDs (stable, rarely change) */
export const APPLE_PODCAST_GENRES: Record<string, string> = {
  "1301": "Arts",
  "1321": "Business",
  "1303": "Comedy",
  "1304": "Education",
  "1483": "Fiction",
  "1511": "Government",
  "1512": "Health & Fitness",
  "1487": "History",
  "1305": "Kids & Family",
  "1502": "Leisure",
  "1310": "Music",
  "1489": "News",
  "1314": "Religion & Spirituality",
  "1533": "Science",
  "1324": "Society & Culture",
  "1545": "Sports",
  "1318": "Technology",
  "1488": "True Crime",
  "1309": "TV & Film",
};

/** Entry from Apple Charts API */
export interface AppleChartEntry {
  id: string;
  name: string;
  artistName: string;
  artworkUrl100: string;
  genres: { genreId: string; name: string; url: string }[];
  url: string;
}

/** Result from iTunes Lookup/Search API */
export interface AppleLookupResult {
  collectionId: number;
  collectionName: string;
  artistName: string;
  feedUrl: string;
  artworkUrl30?: string;
  artworkUrl60?: string;
  artworkUrl100?: string;
  artworkUrl600?: string;
  releaseDate?: string;
  collectionExplicitness?: string;
  trackExplicitness?: string;
  trackCount?: number;
  country?: string;
  primaryGenreName?: string;
  genreIds?: string[];
  genres?: string[];
  contentAdvisoryRating?: string;
}

interface ChartsResponse {
  feed: {
    results: AppleChartEntry[];
  };
}

interface LookupResponse {
  resultCount: number;
  results: AppleLookupResult[];
}

function delay(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

async function fetchWithRetry(url: string, retries = RETRY_MAX): Promise<Response> {
  for (let attempt = 0; attempt < retries; attempt++) {
    const res = await fetch(url);
    if (res.ok) return res;
    if (res.status === 429 || res.status >= 500) {
      if (attempt < retries - 1) {
        await delay(RETRY_BASE_MS * Math.pow(2, attempt));
        continue;
      }
    }
    throw new Error(`Apple API error: ${res.status} ${res.statusText} for ${url}`);
  }
  throw new Error(`Apple API failed after ${retries} retries for ${url}`);
}

export class ApplePodcastsClient {
  /**
   * Fetch top podcasts for a single genre from Apple Charts.
   */
  async topByGenre(
    genreId: string,
    limit: number = 200,
    country: string = "us"
  ): Promise<AppleChartEntry[]> {
    const url = `${CHARTS_BASE}/${country}/podcasts/top/${limit}/genre=${genreId}/podcasts.json`;
    try {
      const res = await fetchWithRetry(url);
      const data = (await res.json()) as ChartsResponse;
      return data.feed?.results ?? [];
    } catch (err) {
      console.warn(`[apple] Failed to fetch genre ${genreId}: ${err}`);
      return [];
    }
  }

  /**
   * Fetch top podcasts across all 19 genres, deduplicated by Apple ID.
   * Merges genre data when a podcast appears in multiple categories.
   */
  async topAllGenres(
    limit: number = 200,
    country: string = "us"
  ): Promise<AppleChartEntry[]> {
    const genreIds = Object.keys(APPLE_PODCAST_GENRES);
    const seen = new Map<string, AppleChartEntry>();

    for (const genreId of genreIds) {
      const entries = await this.topByGenre(genreId, limit, country);
      for (const entry of entries) {
        const existing = seen.get(entry.id);
        if (existing) {
          // Merge genres from duplicate appearance
          const existingGenreIds = new Set(existing.genres.map((g) => g.genreId));
          for (const genre of entry.genres) {
            if (!existingGenreIds.has(genre.genreId)) {
              existing.genres.push(genre);
            }
          }
        } else {
          seen.set(entry.id, { ...entry, genres: [...entry.genres] });
        }
      }
      await delay(INTER_REQUEST_DELAY_MS);
    }

    return Array.from(seen.values());
  }

  /**
   * Batch lookup Apple podcast IDs via iTunes Lookup API.
   * Chunks into groups of 150 IDs per request with retry/backoff.
   */
  async lookupBatch(ids: string[]): Promise<AppleLookupResult[]> {
    const results: AppleLookupResult[] = [];
    const chunks: string[][] = [];

    for (let i = 0; i < ids.length; i += LOOKUP_BATCH_SIZE) {
      chunks.push(ids.slice(i, i + LOOKUP_BATCH_SIZE));
    }

    for (const chunk of chunks) {
      const csvIds = chunk.join(",");
      const url = `${ITUNES_BASE}/lookup?id=${csvIds}&entity=podcast`;
      try {
        const res = await fetchWithRetry(url);
        const data = (await res.json()) as LookupResponse;
        // Filter to only podcast results (lookup can return artist results too)
        const podcasts = data.results.filter(
          (r: any) => r.wrapperType === "track" && r.kind === "podcast"
        );
        results.push(...podcasts);
      } catch (err) {
        console.warn(`[apple] Lookup batch failed (${chunk.length} IDs): ${err}`);
        // Continue with remaining batches — partial results are acceptable
      }
      if (chunks.indexOf(chunk) < chunks.length - 1) {
        await delay(INTER_REQUEST_DELAY_MS);
      }
    }

    return results;
  }

  /**
   * Search iTunes for podcasts by term.
   * Returns results with feedUrl directly (no second lookup needed).
   */
  async search(term: string, limit: number = 25): Promise<AppleLookupResult[]> {
    const url = `${ITUNES_BASE}/search?term=${encodeURIComponent(term)}&media=podcast&limit=${limit}`;
    try {
      const res = await fetchWithRetry(url);
      const data = (await res.json()) as LookupResponse;
      return data.results ?? [];
    } catch (err) {
      console.warn(`[apple] Search failed for "${term}": ${err}`);
      return [];
    }
  }
}
```

```typescript
  identifier: "apple",

  async discover(count, env) {

    const client = new ApplePodcastsClient();

    // Fetch top charts across all genres
    const chartEntries = await client.topAllGenres(count, "us");
    if (chartEntries.length === 0) return [];

    // Batch lookup to get feed URLs + full metadata
    const appleIds = chartEntries.map((e) => e.id);
    const lookupResults = await client.lookupBatch(appleIds);

    // Build lookup map by collectionId
    const lookupMap = new Map(
      lookupResults.map((r) => [String(r.collectionId), r])
    );

    // Merge chart entries with lookup results
    const discovered: DiscoveredPodcast[] = [];
    for (const entry of chartEntries) {
      const lookup = lookupMap.get(entry.id);
      if (!lookup?.feedUrl) continue; // Skip if no feed URL resolved

      discovered.push({
        feedUrl: lookup.feedUrl,
        title: lookup.collectionName || entry.name,
        description: undefined, // iTunes Lookup doesn't return descriptions
        imageUrl: lookup.artworkUrl600 || lookup.artworkUrl100 || entry.artworkUrl100,
        author: lookup.artistName || entry.artistName,
        appleId: entry.id,
        categories: entry.genres.map((g) => ({
          genreId: g.genreId,
          name: g.name,
        })),
        appleMetadata: lookup as unknown as Record<string, unknown>,
      });
    }

    return discovered;
  },

  async search(query, env) {
    const { ApplePodcastsClient } = await import("./apple-podcasts");
    const client = new ApplePodcastsClient();
    const results = await client.search(query);
    return results.map((r) => ({
      feedUrl: r.feedUrl,
      title: r.collectionName,
      imageUrl: r.artworkUrl600 || r.artworkUrl100,
      author: r.artistName,
      appleId: String(r.collectionId),
      categories: (r.genreIds ?? [])
        .filter((id) => id !== "26") // Filter out "Podcasts" meta-genre
        .map((id) => ({ genreId: id, name: r.genres?.[r.genreIds?.indexOf(id) ?? -1] ?? "" })),
```

```typescript
import type { Env } from "../types";
import type { CatalogRefreshMessage } from "../lib/queue-messages";
import { createPrismaClient } from "../lib/db";
import { getCatalogSource } from "../lib/catalog-sources";
import type { DiscoveredPodcast } from "../lib/catalog-sources";

/**
 * Queue consumer for catalog seed and refresh operations.
 * Runs in queue context (15-min limit) to avoid Worker HTTP handler timeout.
 */
export async function handleCatalogRefresh(
  batch: MessageBatch<CatalogRefreshMessage>,
  env: Env,
  ctx: ExecutionContext
): Promise<void> {
  const prisma = createPrismaClient(env.HYPERDRIVE);

  try {
    for (const msg of batch.messages) {
      const { action } = msg.body;
      console.log(`[catalog-refresh] Starting ${action}...`);

      try {
        await updateStatus(prisma, "fetching_charts");

        if (action === "seed") {
          await wipeCatalogData(prisma);
        }

        // Discover podcasts from Apple
        const source = getCatalogSource("apple");
        const discovered = await source.discover(200, env);
        console.log(`[catalog-refresh] Discovered ${discovered.length} unique podcasts`);

        await updateStatus(prisma, "resolving_metadata");

        // Upsert categories
        const categoryIdMap = await upsertCategories(prisma, discovered);

        await updateStatus(prisma, "upserting");

        // Upsert podcasts + join records
        const upsertedIds = await upsertPodcasts(prisma, discovered, categoryIdMap);

        // Handle pending deletion for refresh (not seed)
        if (action === "refresh") {
          await markPendingDeletion(prisma, upsertedIds);
        }

        // Queue feed refresh for all active podcasts
        await queueFeedRefresh(env, upsertedIds);

        await updateStatus(prisma, "complete");
        console.log(`[catalog-refresh] ${action} complete. ${upsertedIds.length} podcasts processed.`);
        msg.ack();
      } catch (err) {
        console.error(`[catalog-refresh] ${action} failed:`, err);
        await updateStatus(prisma, "failed");
        msg.retry();
      }
    }
  } finally {
    ctx.waitUntil(prisma.$disconnect());
  }
}

async function updateStatus(prisma: any, status: string): Promise<void> {
  await prisma.platformConfig.upsert({
    where: { key: "catalogRefresh.status" },
    update: { value: status },
    create: { key: "catalogRefresh.status", value: status, description: "Catalog refresh status" },
  });
}

async function wipeCatalogData(prisma: any): Promise<void> {
  console.log("[catalog-refresh] Wiping existing catalog data...");
  // Delete in dependency order
  await prisma.feedItem.deleteMany({});
  await prisma.briefing.deleteMany({});
  await prisma.briefingRequest.deleteMany({});
  await prisma.clip.deleteMany({});
  await prisma.distillation.deleteMany({});
  await prisma.subscription.deleteMany({});
  await prisma.episode.deleteMany({});
  // Podcast delete cascades to PodcastCategory
  await prisma.podcast.deleteMany({});
  await prisma.category.deleteMany({});
  console.log("[catalog-refresh] Wipe complete.");
}

async function upsertCategories(
  prisma: any,
  discovered: DiscoveredPodcast[]
): Promise<Map<string, string>> {
  // Collect unique genres across all discovered podcasts
  const genreMap = new Map<string, string>(); // genreId → name
  for (const podcast of discovered) {
    for (const cat of podcast.categories ?? []) {
      if (cat.genreId && cat.genreId !== "26") {
        // Skip "Podcasts" meta-genre
        genreMap.set(cat.genreId, cat.name);
      }
    }
  }

  // Upsert each category, build genreId → categoryId map
  const categoryIdMap = new Map<string, string>(); // genreId → cuid
  for (const [genreId, name] of genreMap) {
    const category = await prisma.category.upsert({
      where: { appleGenreId: genreId },
      update: { name },
      create: { appleGenreId: genreId, name },
    });
    categoryIdMap.set(genreId, category.id);
  }

  console.log(`[catalog-refresh] Upserted ${categoryIdMap.size} categories`);
  return categoryIdMap;
}

async function upsertPodcasts(
  prisma: any,
  discovered: DiscoveredPodcast[],
  categoryIdMap: Map<string, string>
): Promise<string[]> {
  const upsertedIds: string[] = [];

  for (const podcast of discovered) {
    if (!podcast.feedUrl) continue;

    const categoryNames = (podcast.categories ?? [])
      .filter((c) => c.genreId !== "26")
      .map((c) => c.name);

    const data = {
      title: podcast.title,
      description: podcast.description,
      imageUrl: podcast.imageUrl,
      author: podcast.author,
      appleId: podcast.appleId,
      podcastIndexId: podcast.podcastIndexId,
      categories: categoryNames,
      appleMetadata: podcast.appleMetadata ?? undefined,
      language: "en", // Default for US storefront; overwritten by feed-refresh
      source: "apple",
      status: "active",
    };

    try {
      // Use feedUrl as the primary upsert key (most stable)
      const upserted = await prisma.podcast.upsert({
        where: { feedUrl: podcast.feedUrl },
        update: {
          ...data,
          // Don't overwrite status if admin has archived it
          status: undefined,
        },
        create: {
          ...data,
          feedUrl: podcast.feedUrl,
        },
      });

      // Auto-restore pending_deletion podcasts that reappear in charts
      if (upserted.status === "pending_deletion") {
        await prisma.podcast.update({
          where: { id: upserted.id },
          data: { status: "active" },
        });
      }

      upsertedIds.push(upserted.id);

      // Update PodcastCategory join records
      const genreIds = (podcast.categories ?? [])
        .filter((c) => c.genreId !== "26")
        .map((c) => c.genreId);

      // Clear existing and recreate (simpler than diffing)
      await prisma.podcastCategory.deleteMany({
        where: { podcastId: upserted.id },
      });

      const joinRecords = genreIds
        .map((genreId) => {
          const categoryId = categoryIdMap.get(genreId);
          if (!categoryId) return null;
          return { podcastId: upserted.id, categoryId };
        })
        .filter(Boolean);

      if (joinRecords.length > 0) {
        await prisma.podcastCategory.createMany({
          data: joinRecords,
          skipDuplicates: true,
        });
      }
    } catch (err) {
      console.warn(`[catalog-refresh] Failed to upsert podcast "${podcast.title}":`, err);
    }
  }

  console.log(`[catalog-refresh] Upserted ${upsertedIds.length} podcasts`);
  return upsertedIds;
}

async function markPendingDeletion(
  prisma: any,
  chartPodcastIds: string[]
): Promise<void> {
  const chartIdSet = new Set(chartPodcastIds);

  // Find active podcasts not in the current chart results
  const activePodcasts = await prisma.podcast.findMany({
    where: { status: "active" },
    select: { id: true },
  });

  const toMark: string[] = [];
  for (const podcast of activePodcasts) {
    if (!chartIdSet.has(podcast.id)) {
      // Check if podcast has active subscriptions
      const subCount = await prisma.subscription.count({
        where: { podcastId: podcast.id },
      });
      if (subCount === 0) {
        toMark.push(podcast.id);
      }
    }
  }

  if (toMark.length > 0) {
    await prisma.podcast.updateMany({
      where: { id: { in: toMark } },
      data: { status: "pending_deletion" },
    });
    console.log(`[catalog-refresh] Marked ${toMark.length} podcasts as pending_deletion`);
  }
}

async function queueFeedRefresh(env: Env, podcastIds: string[]): Promise<void> {
  if (podcastIds.length === 0) return;

  const messages = podcastIds.map((podcastId) => ({
    body: { podcastId, type: "manual" as const },
  }));

  // sendBatch supports up to 100 messages per call
  const BATCH_SIZE = 100;
  for (let i = 0; i < messages.length; i += BATCH_SIZE) {
    const batch = messages.slice(i, i + BATCH_SIZE);
    await env.FEED_REFRESH_QUEUE.sendBatch(batch);
  }

  console.log(`[catalog-refresh] Queued feed refresh for ${podcastIds.length} podcasts`);
}
```

```typescript
});
```
